Here’s the same **Exam** scaffolding, updated so that **`exam_id` is auto-generated** by the database and **not** supplied by the client:

---

### 1. `app/models.py`

```python
# app/models.py
from sqlalchemy import Column, Integer, String, Boolean, Date
from .database import Base

class Exam(Base):
    __tablename__ = "exam"

    exam_id       = Column(Integer, primary_key=True, index=True)  # auto-increments by default
    country       = Column(String,  nullable=False)
    province      = Column(String,  nullable=False)
    answers       = Column(Boolean, nullable=False)
    subject       = Column(String,  nullable=False)
    year          = Column(Date,    nullable=False)
    duration      = Column(Integer, nullable=False)   # in minutes
    coefficient   = Column(Integer, nullable=False)
    exam_path     = Column(String,  nullable=False)
    answers_path  = Column(String,  nullable=True)
```

_No changes needed here beyond noting that primary keys auto-increment unless you override them._

---

### 2. `app/schemas.py`

```python
# app/schemas.py
from pydantic import BaseModel, Field
from datetime import date
from typing import Optional

class ExamBase(BaseModel):
    country:      str           = Field(..., example="China")
    province:     str           = Field(..., example="Guangdong")
    answers:      bool          = Field(..., example=True)
    subject:      str           = Field(..., example="Mathematics")
    year:         date          = Field(..., example="2020-06-01")
    duration:     int           = Field(..., ge=0, example=120)
    coefficient:  int           = Field(..., ge=0, example=2)
    exam_path:    str           = Field(..., example="files/exam1.pdf")
    answers_path: Optional[str] = Field(None, example="files/ans1.pdf")

class ExamCreate(ExamBase):
    """When creating, client DOES NOT supply exam_id."""
    pass

class Exam(ExamBase):
    """What we return in responses—includes the generated ID."""
    exam_id: int

    class Config:
        orm_mode = True
```

- **`ExamCreate`** now has **no** `exam_id` field.
- **`Exam`** still includes `exam_id` so clients see the assigned value.

---

### 3. `app/crud.py`

```python
# app/crud.py
from sqlalchemy.orm import Session
from . import models, schemas

def get_exam(db: Session, exam_id: int) -> models.Exam | None:
    return db.query(models.Exam).filter(models.Exam.exam_id == exam_id).first()

def get_exams(db: Session, skip: int = 0, limit: int = 100):
    return db.query(models.Exam).offset(skip).limit(limit).all()

def create_exam(db: Session, exam: schemas.ExamCreate) -> models.Exam:
    db_ex = models.Exam(**exam.dict())  # no exam_id here
    db.add(db_ex)
    db.commit()
    db.refresh(db_ex)  # now db_ex.exam_id is populated
    return db_ex

def delete_exam(db: Session, exam_id: int) -> bool:
    ex = get_exam(db, exam_id)
    if not ex:
        return False
    db.delete(ex)
    db.commit()
    return True
```

_No need to check for duplicate ID on create, since the DB is generating it._

---

### 4. `app/routes/exam.py`

```python
# app/routes/exam.py
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session
from .. import crud, schemas, database

router = APIRouter(prefix="/exams", tags=["exams"])

def get_db():
    db = database.SessionLocal()
    try:
        yield db
    finally:
        db.close()

@router.post("/", response_model=schemas.Exam, status_code=201)
def create_exam(exam: schemas.ExamCreate, db: Session = Depends(get_db)):
    return crud.create_exam(db, exam)

@router.get("/{exam_id}", response_model=schemas.Exam)
def read_exam(exam_id: int, db: Session = Depends(get_db)):
    db_ex = crud.get_exam(db, exam_id)
    if not db_ex:
        raise HTTPException(status_code=404, detail="Exam not found")
    return db_ex

@router.get("/", response_model=list[schemas.Exam])
def list_exams(skip: int = 0, limit: int = 100, db: Session = Depends(get_db)):
    return crud.get_exams(db, skip, limit)

@router.delete("/{exam_id}", status_code=204)
def delete_exam(exam_id: int, db: Session = Depends(get_db)):
    if not crud.delete_exam(db, exam_id):
        raise HTTPException(status_code=404, detail="Exam not found")
```

- The **POST** endpoint no longer expects `exam_id` in the payload.

### 5. `app/main.py`

```python
# app/main.py
from fastapi import FastAPI
from . import models, database
from .routes import person, exam

# Create tables (including auto-incr PKs)
models.Base.metadata.create_all(bind=database.engine)

app = FastAPI(title="Exam Collection API")

app.include_router(person.router)
app.include_router(exam.router)
```

---

Now:

- **Clients** send everything **except** `exam_id` to **POST** `/exams/`.  
- The **database** auto-assigns `exam_id`.  
- Responses include the newly generated `exam_id`.
